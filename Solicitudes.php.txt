<script>
let solicitudes = JSON.parse(localStorage.getItem("solicitudes")) || [];

// Helpers
const hoy = () => new Date().toLocaleString();

// Render
function render(lista = solicitudes){
  const tbody = document.getElementById("tbody");
  tbody.innerHTML = "";

  lista.forEach((s,i)=>{
    tbody.innerHTML += `
      <tr>
        <td>${s.nombre}</td>
        <td>${s.asunto}</td>
        <td>${s.tipo}</td>
        <td><span class="badge ${s.prioridad}">${s.prioridad}</span></td>
        <td><span class="badge ${s.estado}">${s.estado.replace("_"," ")}</span></td>
        <td>${s.fecha}</td>
        <td class="acciones">
          <button class="btn-state" onclick="cambiarEstado(${i})">Estado</button>
          <button class="btn-edit" onclick="editar(${i})">Editar</button>
          <button class="btn-del" onclick="eliminar(${i})">Eliminar</button>
        </td>
      </tr>
    `;
  });

  kpis();
}

// KPIs
function kpis(){
  const t = solicitudes.length;
  const p = solicitudes.filter(s=>s.estado==="Pendiente").length;
  const e = solicitudes.filter(s=>s.estado==="En_proceso").length;
  const r = solicitudes.filter(s=>s.estado==="Resuelta").length;

  document.getElementById("kpiTotal").innerText = t;
  document.getElementById("kpiPend").innerText = p;
  document.getElementById("kpiProc").innerText = e;
  document.getElementById("kpiRes").innerText = r;
}

// Guardar
function save(){
  localStorage.setItem("solicitudes", JSON.stringify(solicitudes));
}

// Crear/Actualizar
document.getElementById("formSolicitud").addEventListener("submit", e=>{
  e.preventDefault();

  const id = document.getElementById("solId").value;

  const data = {
    nombre: nombre.value,
    correo: correo.value,
    asunto: asunto.value,
    tipo: tipo.value,
    prioridad: prioridad.value,
    descripcion: descripcion.value,
    estado: "Pendiente",
    fecha: hoy()
  };

  if(id===""){
    solicitudes.push(data);
  } else {
    data.estado = solicitudes[id].estado;
    data.fecha = solicitudes[id].fecha;
    solicitudes[id] = data;
  }

  save();
  render();
  e.target.reset();
  solId.value="";
});

// Acciones
function cambiarEstado(i){
  const estados = ["Pendiente","En_proceso","Resuelta"];
  let idx = estados.indexOf(solicitudes[i].estado);
  solicitudes[i].estado = estados[(idx+1)%3];
  save(); render();
}

function editar(i){
  const s = solicitudes[i];
  solId.value = i;
  nombre.value = s.nombre;
  correo.value = s.correo;
  asunto.value = s.asunto;
  tipo.value = s.tipo;
  prioridad.value = s.prioridad;
  descripcion.value = s.descripcion;
}

function eliminar(i){
  if(confirm("¿Eliminar solicitud?")){
    solicitudes.splice(i,1);
    save(); render();
  }
}

// Filtros
function buscar(){
  const txt = document.getElementById("buscar").value.toLowerCase();
  const lista = solicitudes.filter(s =>
    s.nombre.toLowerCase().includes(txt) ||
    s.asunto.toLowerCase().includes(txt)
  );
  render(lista);
}

function filtrar(){
  let estado = fEstado.value;
  let prioridad = fPrioridad.value;

  let lista = solicitudes.filter(s =>
    (estado==="todas" || s.estado===estado) &&
    (prioridad==="todas" || s.prioridad===prioridad)
  );

  render(lista);
}

// Exportar CSV
function exportarCSV(){
  let csv = "Nombre,Correo,Asunto,Tipo,Prioridad,Estado,Fecha\n";
  solicitudes.forEach(s=>{
    csv += `${s.nombre},${s.correo},${s.asunto},${s.tipo},${s.prioridad},${s.estado},${s.fecha}\n`;
  });

  const blob = new Blob([csv], {type:"text/csv"});
  const a = document.createElement("a");
  a.href = URL.createObjectURL(blob);
  a.download = "solicitudes.csv";
  a.click();
}

// Init
render();
